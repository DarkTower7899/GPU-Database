function showSimilarCards(baseCard) { const resultsSection = document.getElementById('resultsSection'); const resultsTitle = document.getElementById('resultsTitle'); const resultsCount = document.getElementById('resultsCount'); const gpuList = document.getElementById('gpuList'); let similar = GPU_DB .map(gpu => { const diff = gpu.score - baseCard.score; const pctDiff = ((diff / baseCard.score) * 100).toFixed(1); return { ...gpu, diff, pctDiff, isBase: gpu.name === baseCard.name }; }) .sort((a, b) => a.diff - b.diff); const filtered = similar.filter(g => !g.isBase); // Corrected sorting logic: positive diff = faster, negative = slower const faster = filtered.filter(g => g.diff > 0).sort((a, b) => b.diff - a.diff).slice(0, 7); const slower = filtered.filter(g => g.diff < 0).sort((a, b) => a.diff - b.diff).slice(0, 8); // 🔽 UPDATED: Place the searched GPU at the TOP of the list const finalList = [baseCard, ...faster, ...slower]; resultsSection.classList.remove('hidden'); resultsTitle.textContent = `Similar to: ${baseCard.name}`; resultsCount.textContent = `${finalList.length} cards shown`; gpuList.innerHTML = finalList.map((gpu, i) => { const isBase = gpu.isBase; const isFaster = gpu.diff > 0; const isSlower = gpu.diff < 0; const cardClass = isBase ? 'base-card' : (isFaster ? 'faster' : 'slower'); const diffClass = isFaster ? 'positive' : 'negative'; const diffIcon = isFaster ? '+' : ''; const rankClass = i < 3 ? 'top3' : ''; return `

${gpu.name} ${isBase ? '(BASE)' : ''}

${gpu.arch} ${gpu.year} ${gpu.vram}
DX:${gpu.dx_hw} VK:${gpu.vulkan} GL:${gpu.opengl}
#${i + 1}
${diffIcon}${gpu.pctDiff}%
Diff
${gpu.score.toLocaleString()}
Score
${gpu.tdp}W • ${gpu.brand.toUpperCase()}
`; }).join(''); showStatus(`Found ${finalList.length} cards with similar performance to ${baseCard.name}. Click a card to compare.`, 'done'); }
